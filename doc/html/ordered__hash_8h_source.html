179 /**
180  * Internal common class used by ordered_map and ordered_set.
181  *
182  * ValueType is what will be stored by ordered_hash (usually std::pair<Key, T> for map and Key for set).
183  *
184  * KeySelect should be a FunctionObject which takes a ValueType in parameter and return a reference to the key.
185  *
186  * ValueSelect should be a FunctionObject which takes a ValueType in parameter and return a reference to the value.
187  * ValueSelect should be void if there is no value (in set for example).
188  *
189  * ValueTypeContainer is the container which will be used to store ValueType values.
190  * Usually a std::deque<ValueType, Allocator> or std::vector<ValueType, Allocator>.
191  *
192  *
193  *
194  * The orderd_hash structure is a hash table which preserves the order of insertion of the elements.
195  * To do so, it stores the values in the ValueTypeContainer (m_values) using emplace_back at each
196  * insertion of a new element. Another structure (m_buckets of type std::vector<bucket_entry>) will
197  * serve as buckets array for the hash table part. Each bucket stores an index which corresponds to
198  * the index in m_values where the bucket's value is and the (truncated) hash of this value. An index
199  * is used instead of a pointer to the value to reduce the size of each bucket entry.
200  *
201  * To resolve collisions in the buckets array, the structures use robin hood linear probing with
202  * backward shift deletion.
203  */

556  iterator erase(const_iterator pos) {
557  tsl_assert(pos != cend());
558 
559  const std::size_t index_erase = iterator_to_index(pos);
560 
561  auto it_bucket = find_key(pos.key(), hash_key(pos.key()));
562  tsl_assert(it_bucket != m_buckets.end());
563 
564  erase_value_from_bucket(it_bucket);
565 
566  /*
567  * One element was removed from m_values, due to the left shift the next element
568  * is now at the position of the previous element (or end if none).
569  */
570  return begin() + index_erase;
571  }
572 
573  iterator erase(const_iterator first, const_iterator last) {
574  if(first == last) {
575  return mutable_iterator(first);
576  }
577 
578  tsl_assert(std::distance(first, last) > 0);
579  const std::size_t start_index = iterator_to_index(first);
580  const std::size_t nb_values = std::size_t(std::distance(first, last));
581  const std::size_t end_index = start_index + nb_values;
582 
583  // Delete all values
584 #ifdef TSL_NO_CONTAINER_ERASE_CONST_ITERATOR
585  auto next_it = m_values.erase(mutable_iterator(first).m_iterator, mutable_iterator(last).m_iterator);
586 #else
587  auto next_it = m_values.erase(first.m_iterator, last.m_iterator);
588 #endif
589 
590  /*
591  * Mark the buckets corresponding to the values as empty and do a backward shift.
592  *
593  * Also, the erase operation on m_values has shifted all the values on the right of last.m_iterator.
594  * Adapt the indexes for these values.
595  */
596  std::size_t ibucket = 0;
597  while(ibucket < m_buckets.size()) {
598  if(m_buckets[ibucket].empty()) {
599  ibucket++;
600  }
601  else if(m_buckets[ibucket].index() >= start_index && m_buckets[ibucket].index() < end_index) {
602  m_buckets[ibucket].clear();
603  backward_shift(ibucket);
604  // Don't increment ibucket, backward_shift may have replaced current bucket.
605  }
606  else if(m_buckets[ibucket].index() >= end_index) {
607  m_buckets[ibucket].set_index(index_type(m_buckets[ibucket].index() - nb_values));
608  ibucket++;
609  }
610  else {
611  ibucket++;
612  }
613  }
614 
615  return iterator(next_it);
616  }

956  *
957  * From the bucket_for_hash, search for the value until we either find an empty bucket
958  * or a bucket which has a value with a distance from its initial bucket longer

980  void rehash_impl(size_type bucket_count) {
981  bucket_count = round_up_to_power_of_two(bucket_count);
982  tsl_assert(bucket_count > 0);
983 
984  if(bucket_count == this->bucket_count()) {
985  return;
986  }
987 
988  if(bucket_count > max_bucket_count()) {
989  throw std::length_error("The map exceeds its maxmimum size.");
990  }
991 
992 
993  buckets_container_type old_buckets(bucket_count);
994  m_buckets.swap(old_buckets);
995  // Everything should be noexcept from here.
996 
997  m_mask = bucket_count - 1;
998  this->max_load_factor(m_max_load_factor);
999  m_grow_on_next_insert = false;
1000 
1001 
1002 
1003  for(const bucket_entry& old_bucket: old_buckets) {
1004  if(old_bucket.empty()) {
1005  continue;
1006  }
1007 
1008  truncated_hash_type insert_hash = old_bucket.truncated_hash();
1009  index_type insert_index = old_bucket.index();
1010 
1011  for(std::size_t ibucket = bucket_for_hash(insert_hash), dist_from_init_bucket = 0; ;
1012  ibucket = next_bucket(ibucket), dist_from_init_bucket++)
1013  {
1014  if(m_buckets[ibucket].empty()) {
1015  m_buckets[ibucket].set_index(insert_index);
1016  m_buckets[ibucket].set_hash(insert_hash);
1017  break;
1018  }
1019 
1020  const std::size_t distance = dist_from_initial_bucket(ibucket);
1021  if(dist_from_init_bucket > distance) {
1022  std::swap(insert_index, m_buckets[ibucket].index_ref());
1023  std::swap(insert_hash, m_buckets[ibucket].truncated_hash_ref());
1024  dist_from_init_bucket = distance;
1025  }
1026  }
1027  }
1028  }
