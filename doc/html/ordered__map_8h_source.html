467  iterator erase(const_iterator first, const_iterator last) {
468  if(first == last) {
469  return mutable_iterator(first);
470  }
471 
472  tsl_assert(std::distance(first, last) > 0 && std::distance(cbegin(), first) >= 0);
473  const std::size_t start_index = std::size_t(std::distance(cbegin(), first));
474  const std::size_t nb_values = std::size_t(std::distance(first, last));
475  const std::size_t end_index = start_index + nb_values;
476 
477  // Delete all values
478 #ifdef TSL_NO_CONTAINER_ERASE_CONST_ITERATOR
479  auto next_it = m_values.erase(mutable_iterator(first).m_iterator, mutable_iterator(last).m_iterator);
480 #else
481  auto next_it = m_values.erase(first.m_iterator, last.m_iterator);
482 #endif
483 
484  /*
485  * TODO optimize
486  *
487  * Mark the buckets corresponding to the values as empty and do a backward shift.
488  *
489  * Also, the erase operation on m_values has shifted all the values on the right of last.m_iterator.
490  * Adapt the indexes for these values.
491  */
492  for(std::size_t ibucket = 0; ibucket < m_buckets.size(); ibucket++) {
493  if(m_buckets[ibucket].empty()) {
494  continue;
495  }
496 
497  if(m_buckets[ibucket].index() >= start_index && m_buckets[ibucket].index() < end_index) {
498  m_buckets[ibucket].clear();
499  backward_shift(ibucket);
500  }
501  else if(m_buckets[ibucket].index() >= end_index) {
502  m_buckets[ibucket].set_index(m_buckets[ibucket].index() - nb_values);
503  }
504  }

718  if(it_bucket_key == m_buckets.end()) {

870  void erase_value_from_bucket(typename buckets_container_type::iterator it_bucket) {
871  tsl_assert(it_bucket != m_buckets.end() && !it_bucket->empty());
872 
873  m_values.erase(m_values.begin() + it_bucket->index());
874 
875  const std::size_t index_deleted = it_bucket->index();
876 
877  // m_values.erase shifted all the values on the right of the erased value, shift the indexes except if
878  // it was the last value
879  if(index_deleted != m_values.size()) {
880  for(auto& bucket: m_buckets) {
881  if(!bucket.empty() && bucket.index() > index_deleted) {

883  }
884  }

887  // Mark the bucket as empty and do a backward shift of the values on the right
888  it_bucket->clear();
889  backward_shift(std::size_t(std::distance(m_buckets.begin(), it_bucket)));
890  }

909  /**
910  * From ibucket, search for an empty bucket to store the insert_index and the insert_hash.
911  *
912  * If on the way we find a bucket with a value which is further away from its initial bucket
913  * than our current probing, swap the indexes and the hashes and continue the search
914  * for an empty bucket to store this new index and hash while continuing the swapping process.
915  */
916  void insert_with_robin_hood_swap(std::size_t ibucket, std::size_t iprobe,
917  typename bucket_entry::index_type insert_index,
918  typename bucket_entry::truncated_hash_type insert_hash) noexcept
919  {
920  while(true) {
921  if(m_buckets[ibucket].empty()) {
922  m_buckets[ibucket].set_index(insert_index);
923  m_buckets[ibucket].set_hash(insert_hash);
924 
925  return;
926  }
927 
928  const std::size_t distance = dist_from_initial_bucket(ibucket);
929  if(iprobe > distance) {
930  const auto tmp_index = m_buckets[ibucket].index();
931  const auto tmp_hash = m_buckets[ibucket].truncated_hash();
932 
933  m_buckets[ibucket].set_index(insert_index);
934  m_buckets[ibucket].set_hash(insert_hash);
935 
936  insert_index = tmp_index;
937  insert_hash = tmp_hash;
938 
939  iprobe = distance;
940  }
941 
942  ibucket = next_probe(ibucket);
943  ++iprobe;
944  }
945  }

947  std::size_t dist_from_initial_bucket(std::size_t ibucket) const noexcept {
948  const std::size_t initial_bucket = bucket_for_hash(m_buckets[ibucket].truncated_hash());
949 
950  // If the bucket is smaller than the initial bucket for the value, there was a wrapping at the end of the
951  // bucket array due to the modulo.
952  if(ibucket < initial_bucket) {
953  return (bucket_count() + ibucket) - initial_bucket;
954  }
955  else {
956  return ibucket - initial_bucket;
957  }

973  std::pair<iterator, bool> insert_impl(K&& key, std::size_t hash, Args&&... value) {
974  resize_if_needed(1);
975 
976  for(std::size_t ibucket = bucket_for_hash(hash), iprobe = 0; ; ibucket = next_probe(ibucket), ++iprobe) {
977  if(m_buckets[ibucket].empty()) {
978  emplace_back_values_container(std::forward<K>(key), std::forward<Args>(value)...);
979  m_buckets[ibucket].set_index(m_values.size() - 1);
980  m_buckets[ibucket].set_hash(hash);

983  }
984  else if(m_buckets[ibucket].truncated_hash() == bucket_entry::truncate_hash(hash) &&
985  compare_keys(key, KeySelect()(m_values[m_buckets[ibucket].index()])))
986  {
987  return std::make_pair(begin() + m_buckets[ibucket].index(), false);
988  }
989  else if(rehash_on_high_nb_probes(iprobe)) {
990  return insert_impl(std::forward<K>(key), hash, std::forward<Args>(value)...);
991  }
992  else {
993  const std::size_t distance = dist_from_initial_bucket(ibucket);
994  if(iprobe > distance) {
995  emplace_back_values_container(std::forward<K>(key), std::forward<Args>(value)...);
996 
997  // Propagate the index and the hash of the current bucket to a more far away bucket
998  // Clear the current bucket so we can use it to insert the key.
999  insert_with_robin_hood_swap(next_probe(ibucket), distance + 1,
1000  m_buckets[ibucket].index(), m_buckets[ibucket].truncated_hash());
1001 
1002  m_buckets[ibucket].set_index(m_values.size() - 1);
1003  m_buckets[ibucket].set_hash(hash);
1004 
1005  return std::make_pair(std::prev(end()), true);
1006  }
1007  }
1008  }
