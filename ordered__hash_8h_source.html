197 /**
198  * Internal common class used by ordered_map and ordered_set.
199  *
200  * ValueType is what will be stored by ordered_hash (usually std::pair<Key, T> for map and Key for set).
201  *
202  * KeySelect should be a FunctionObject which takes a ValueType in parameter and return a reference to the key.
203  *
204  * ValueSelect should be a FunctionObject which takes a ValueType in parameter and return a reference to the value.
205  * ValueSelect should be void if there is no value (in set for example).
206  *
207  * ValueTypeContainer is the container which will be used to store ValueType values.
208  * Usually a std::deque<ValueType, Allocator> or std::vector<ValueType, Allocator>.
209  *
210  *
211  *
212  * The orderd_hash structure is a hash table which preserves the order of insertion of the elements.
213  * To do so, it stores the values in the ValueTypeContainer (m_values) using emplace_back at each
214  * insertion of a new element. Another structure (m_buckets of type std::vector<bucket_entry>) will
215  * serve as buckets array for the hash table part. Each bucket stores an index which corresponds to
216  * the index in m_values where the bucket's value is and the (truncated) hash of this value. An index
217  * is used instead of a pointer to the value to reduce the size of each bucket entry.
218  *
219  * To resolve collisions in the buckets array, the structures use robin hood linear probing with
220  * backward shift deletion.
221  */

1060  * or a bucket which has a value with a distance from its ideal bucket longer

1082  void rehash_impl(size_type bucket_count) {
1083  tsl_assert(bucket_count >= size_type(std::ceil(float(size())/max_load_factor())));
1084 
1085  if(bucket_count > max_bucket_count()) {
1086  TSL_THROW_OR_TERMINATE(std::length_error("The map exceeds its maxmimum size."));
1087  }
1088 
1089  if(bucket_count > 0) {
1090  bucket_count = round_up_to_power_of_two(bucket_count);
1091  }
1092 
1093  if(bucket_count == this->bucket_count()) {
1094  return;
1095  }
1096 
1097 
1098  buckets_container_type old_buckets(bucket_count);
1099  m_buckets.swap(old_buckets);
1100  m_first_or_empty_bucket = m_buckets.data();
1101  // Everything should be noexcept from here.
1102 
1103  m_mask = (bucket_count > 0)?(bucket_count - 1):0;
1104  this->max_load_factor(m_max_load_factor);
1105  m_grow_on_next_insert = false;
1106 
1107 
1108 
1109  for(const bucket_entry& old_bucket: old_buckets) {
1110  if(old_bucket.empty()) {
1111  continue;
1112  }
1113 
1114  truncated_hash_type insert_hash = old_bucket.truncated_hash();
1115  index_type insert_index = old_bucket.index();
1116 
1117  for(std::size_t ibucket = bucket_for_hash(insert_hash), dist_from_ideal_bucket = 0; ;
1118  ibucket = next_bucket(ibucket), dist_from_ideal_bucket++)
1119  {
1120  if(m_buckets[ibucket].empty()) {
1121  m_buckets[ibucket].set_index(insert_index);
1122  m_buckets[ibucket].set_hash(insert_hash);
1123  break;
1124  }
1125 
1126  const std::size_t distance = distance_from_ideal_bucket(ibucket);
1127  if(dist_from_ideal_bucket > distance) {
1128  std::swap(insert_index, m_buckets[ibucket].index_ref());
1129  std::swap(insert_hash, m_buckets[ibucket].truncated_hash_ref());
1130  dist_from_ideal_bucket = distance;
1131  }
1132  }
1133  }
